\section{System Architecture}
%------------------------------------------------

\begin{frame}{SYSTEM OVERVIEW - The Big Picture}
    \begin{block}{Our 4-Component Smart Detection System}
        \begin{enumerate}
            \item \textbf{LSTM Neural Network} - Forecaster (predicts expected load)
            \item \textbf{K-Means Clustering} - Benchmark (defines normal patterns)
            \item \textbf{Dynamic Programming} - Detection (finds attack timing)
            \item \textbf{Statistical Testing} - Validation (confirms attacks)
        \end{enumerate}
    \end{block}
    
    \begin{alertblock}{Key Insight}
        Each component solves ONE specific problem $\rightarrow$ Together, robust detection

\begin{frame}{COMPONENT 2 - K-Means Clustering}
    \begin{columns}[c]
        \column{0.5\textwidth}
        \begin{block}{The Problem}
            Days have different load shapes:
            \begin{itemize}
                \item Summer: High afternoon (AC)
                \item Winter: Dual peaks (heating)
                \item Weekend: Lower, flatter
            \end{itemize}
        \end{block}
        
        \begin{block}{Solution: K-Means}
            \begin{itemize}
                \item Finds \textbf{typical daily patterns}
                \item Groups similar days
                \item Creates 5 benchmark profiles
            \end{itemize}
        \end{block}
        
        \begin{alertblock}{Why 5 Clusters?}
            \small
            Too few: Misses variations $\mid$ Too many: Overfits $\mid$ \textbf{5 = sweet spot}
        \end{alertblock}
        
        \column{0.5\textwidth}
        \begin{center}

        \end{center}

\end{frame}

\begin{frame}{COMPONENT 4 - Statistical Testing}
    \begin{block}{The Phase 4 Innovation}
        Problem: \textbf{Natural spikes competed with weak attacks}
    \end{block}
    
    \begin{columns}[c]
        \column{0.6\textwidth}
        \begin{block}{Solution: Hypothesis Testing}
            \small
            "Is this statistically UNUSUAL?"
            \begin{enumerate}
                \item Model normal baseline: $\mu = 8\%$, $\sigma = 3\%$
                \item Calculate p-value for each interval
                \item If p < 0.01 $\rightarrow$ ATTACK!
            \end{enumerate}
        \end{block}
        
        \begin{table}
            \centering
            \footnotesize
            \begin{tabular}{lccc}
                \toprule
                \textbf{Method} & \textbf{Natural} & \textbf{Weak Attack} & \textbf{Result} \\
                \midrule
                DP Only & Detected & Missed & FAILS \\
                Statistical & Filtered & Detected & SUCCESS \\
                \textbf{Hybrid} & Filtered & Detected & \textbf{BEST} \\
                \bottomrule
            \end{tabular}
        \end{table}
        
        \column{0.4\textwidth}
        \begin{alertblock}{Safeguards}
            \begin{itemize}
                \item \textbf{Bonferroni}: Multiple testing
                \item \textbf{Cohen's d}: Practical significance
            \end{itemize}
        \end{alertblock}
    \end{columns}
\end{frame}
